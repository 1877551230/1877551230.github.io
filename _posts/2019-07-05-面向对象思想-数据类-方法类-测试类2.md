要求:输入三门课的成绩并计算平均分  
用面向对象编程思想来做:  
* 一个类储存属性并提供修改的方法
* 一个类提供方法功能(求平均数)
* 一个类用来测试


1. 首先创建Score类
Score.java

```java
package cn.tedu.day06.demo5;
/**
 * 专门用来存储和读取数据
 * @author PC
 *
 */
public class Score {
	private int java;
	private int html;
	private int sql;
	public Score(){
		
	}
	public Score(int java, int html, int sql) {
		this.java = java;
		this.html = html;
		this.sql = sql;
	}
	public int getJava() {
		return java;
	}
	public void setJava(int java) {
		this.java = java;
	}
	public int getHtml() {
		return html;
	}
	public void setHtml(int html) {
		this.html = html;
	}
	public int getSql() {
		return sql;
	}
	public void setSql(int sql) {
		this.sql = sql;
	}
	

}

```

2. 创建ScoreOperate类

```java
package cn.tedu.day06.demo5;
/**
 * 专门用来实现成绩的操作
 * 此类没有属性
 * @author PC
 *
 */
public class ScoreOperate {
	public int sum(Score score){
		return score.getJava()+score.getHtml()+score.getSql();
	}
	public double avg(Score score){
		int sum = sum(score);
		return sum/3.0;
	}

}

```

3. 创建测试类Demo1.java

```java
package cn.tedu.day06.demo5;

import java.util.Scanner;

public class Demo1 {

	public static void main(String[] args) {
		Scanner input = new Scanner(System.in);
		System.out.println("请输入java成绩:");
		int java = input.nextInt();
		System.out.println("请输入html成绩:");
		int html = input.nextInt();
		System.out.println("请输入sql成绩:");
		int sql = input.nextInt();
		Score score = new Score(java,html,sql);
		ScoreOperate so = new ScoreOperate();
		double avg = so.avg(score);
		System.out.println(avg);

	}

}
```

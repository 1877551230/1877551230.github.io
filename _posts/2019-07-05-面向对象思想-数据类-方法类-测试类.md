面向对象编程一般遵从该原则:
  * 一个类用来修改数据
  * 一个类用来实现功能
  * 一个类进行测试

要求: 循环输入多个会员的编号，根据会员编号，查找会员积分

1. 首先创建会员类User
User.java

```java
public class User {
	private String id;
	private int point;
	public String getId() {
		return id;
	}
	public void setId(String id) {
		this.id = id;
	}
	public int getPoint() {
		return point;
	}
	public void setPoint(int point) {
		this.point = point;
	}
	
}
```

2. 创建功能类UserOperate   
UserOperate.java  

```java
import java.util.Scanner;

public class UserOperate2 {
	private User[] users = new User[20];//声明类数组,用于存储数据
	public void addUsers(int index){
		Scanner input = new Scanner(System.in);
		System.out.println("请输入会员编号:");
		String id = input.next();
		System.out.println("请输入会员积分:");
		int point = input.nextInt();
		User user = new User();
		user.setId(id);
		user.setPoint(point);
		users[index] = user;
	}
	public void showUsers(){
		for(User user:users){
			if(user!=null)//为了防止输入的数据不到20出现空指针异常,只输出不是null的数据
				System.out.println("id:"+user.getId()+"\tpoint:"+user.getPoint());;
		}
	}
	public int findUserAgeByName(String id){
		int point = -1;
		for(User user:users){
			if(user!=null&&user.getId().equals(id)){
				point = user.getPoint();
			}
		}
		return point;
	}

}
```
